150    def get_screenshot(self, name: str, img_format: str, width: int = None, height: int = None, quality: int = -1) -> bytes:
151        """
152        Retrieves a screenshot of a source.
153
154        :param name: Name of the source to take a screenshot of
155        :param format: Image compression format to use. Use GetVersion to get compatible image formats
156        :param width: Width to scale the screenshot to (>= 8, <= 4096). If not specified, full resolution will be used.
157        :param height: Height to scale the screenshot to (>= 8, <= 4096). If not specified, full resolution will be used.
158        :param quality: Compression quality to use. 0 for high compression, 100 for uncompressed. -1 to use "default"
159        :return: A byte array containing the screenshot data, or None if the screenshot cannot be retrieved.
160        """
161        screenshot_data = self.client.get_source_screenshot(name, img_format, width, height, quality)
162        return screenshot_data.image_data

164    def save_source_screenshot(self, source_name: str, image_format: str, image_file_path: str, image_width: int = None, image_height: int = None, image_compression_quality: int = -1) -> bool:
165        """
166        Saves a screenshot of a source to a file.
167        
168        :param source_name: Name of the source to take a screenshot of
169        :param image_format: Image compression format to use. Use GetVersion to get compatible image formats
170        :param image_file_path: Path to the file to save the screenshot to
171        :param image_width: Width to scale the screenshot to (>= 8, <= 4096). If not specified, full resolution will be used.
172        :param image_height: Height to scale the screenshot to (>= 8, <= 4096). If not specified, full resolution will be used.
173        :param image_compression_quality: Compression quality to use. 0 for high compression, 100 for uncompressed. -1 to use "default"
174        :return: True if the screenshot was saved successfully, False otherwise.
175        """
176        image_data = self.client.save_source_screenshot(source_name, image_format, image_file_path, image_width, image_height, image_compression_quality)
177        with open(image_file_path, "wb") as f:
178            f.write(base64.b64decode(image_data))
179        return True
